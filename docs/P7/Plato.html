def insertar (ali, gr)

        @alimentos.insert_tail(ali[i])
    end

    for i in (0...gr.size) do
        @gramos.insert_tail(gr[i])
    end
end

def por_lipidos
    gr = @gramos.collect{|x| x}
    lip = @alimentos.collect{|x| x.lipidos * 4}
    v_ener = @alimentos.collect{|x| x.valorener}

    v_ener_total = 0
    lipidos_total = 0

    for i in (0...v_ener.size)
        v_ener_total += (v_ener[i]*gr[i])
        lipidos_total += (lip[i]*gr[i])
    end
    
    @por_lip = lipidos_total * 100 / v_ener_total
end

def por_proteinas
    gr = @gramos.collect{|x| x}
    prot = @alimentos.collect{|x| x.proteinas * 4}
    v_ener = @alimentos.collect{|x| x.valorener}

    v_ener_total = 0
    proteinas_total = 0

    for i in (0...v_ener.size)
        v_ener_total += (v_ener[i]*gr[i])
        proteinas_total += (prot[i]*gr[i])
    end
    
    @por_prot = proteinas_total * 100 / v_ener_total
end

def to_s 
    aux1 = @alimentos.head
    aux2 = @gramos.head

    cadena = ""

    while(aux1 != nil)
        cadena << "#{aux1.value.nombre} #{aux2.value}gr, "
        aux1 = aux1.next
        aux2 = aux2.next
    end
    cadena
end

def vct
