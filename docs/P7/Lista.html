def insert_head (contenido)
	aux = Node.new(contenido,nil,nil)
	if (@head == nil)
		@head = aux
		@tail = aux
	else
		aux.next = @head
		@head.prev = aux
		@head = aux
	end
	@size += 1
end

def to_s
	aux = @head
	cadena = "("

	while(aux != nil)
		cadena << "#{aux.value},"
		aux = aux.next
	end

	cadena << ")"
	cadena
end
