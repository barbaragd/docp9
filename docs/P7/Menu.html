def huellanut
    v_ener = @alimentos.collect{|x| x.valorener}
    h_carb = @alimentos.collect{|x| x.gei}
    gram = @gramos.collect{|x| x*0.01}


    v_ener.zip(gram).map{|x,y| x*y}
    h_carb.zip(gram).map{|x,y| x*y}

    vt = 0
    ht = 0

    v_ener.each{|x| vt+=x}
    h_carb.each{|x| ht+=x}

    if vt < 800
        hnc = 1
    elsif vt > 1200
        hnc = 3
    else
        hnc = 2
    end
    
    if ht < 670
        hne = 1
    elsif ht > 830
        hne = 3
    else 
        hne = 2
    end

    huellanutricional = (hne + hnc) / 2.0

            
end

def total_gei
    gases = @alimentos.collect{|x| x.gei / 365}
    gr = @gramos.collect{|x| x}

    gases_total = 0

    for i in (0...gases.size)
        gases_total += (gases[i]*gr[i])
    end
    @gei_diario = gases_total
end

def total_terreno
    terr = @alimentos.collect{|x| x.terreno}
    gr = @gramos.collect{|x| x}

    terreno_total = 0

    for i in (0...terr.size)
        terreno_total += (terr[i]*gr[i])
    end
    @terreno = terreno_total
end
